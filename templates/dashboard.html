實體感測器串接指引

本平台已內建標準 HTTP REST API。您可以讓您的實體 IoT 硬體裝置（如 ESP32、ESP8266 或 Raspberry Pi）定時透過 WiFi 發送 HTTP POST 請求來同步實體量測數據。

API 上報端點 URL： POST http://<your-server-ip>/api/iot/report/
JSON 資料負載格式： 
{
  "pond_id": 1, 
  "sensor_type": "temperature",  // 支援: temperature, ph, dissolved_oxygen, water_level
  "value": 26.85
}

Python 快速測試腳本

import requests
import json

url = "http://127.0.0.1:8000/api/iot/report/"
payload = {
    "pond_id": 1,
    "sensor_type": "temperature",
    "value": 26.85
}
headers = {'Content-Type': 'application/json'}

response = requests.post(url, data=json.dumps(payload), headers=headers)
print("伺服器回應:", response.json())

ESP32 Arduino 實體連網代碼

#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
const char* serverName = "http://<your-server-ip>/api/iot/report/";

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/json");
    
    // 讀取實體溫度感測器，並封裝成 JSON 格式
    String jsonPayload = "{\"pond_id\":1,\"sensor_type\":\"temperature\",\"value\":25.8}";
    
    int httpResponseCode = http.POST(jsonPayload);
    Serial.print("HTTP Response Code: ");
    Serial.println(httpResponseCode);
    http.end();
  }
  delay(10000); // 每 10 秒上報一次
}
